    {String privateKey}) {

  _preference = preference;
  privateKey ??= preference.privateKey;
  atLookUp = AtLookupImpl(atSign, preference.rootDomain, preference.rootPort,
      privateKey: privateKey, cramSecret: preference.cramSecret);
}
