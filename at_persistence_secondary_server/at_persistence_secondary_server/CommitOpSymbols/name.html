String get name {
  switch (this) {
    case CommitOp.UPDATE:
      return '+';
    case CommitOp.UPDATE_META:
      return '#';
    case CommitOp.UPDATE_ALL:
      return '*';
    case CommitOp.DELETE:
      return '-';
    default:
      return null;
  }
}
